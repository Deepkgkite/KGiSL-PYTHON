from math import sqrt


def distance_between(pointA, pointB):
    """Computes the distance between two Point objects.
    
    """
    x1, y1 = pointA
    x2, y2 = pointB

    distx = x1 - x2
    disty = y1 - y2

    return sqrt(distx**2 + disty**2)
